        {
            
        }
        public Employee(int id, string name, int age)
        {
            Id = id;
            Age = age;
            Name = name;
        }
        public override string ToString()
        {
            return "Employee Id : " + Id +
                "\nEmployee Name : " + Name +

        {
            Employee e1, e2;
            e1 = this;
